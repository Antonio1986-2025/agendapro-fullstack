========================================
🚀 COMANDOS RÁPIDOS PARA VPS
========================================

COPIE E COLE ESTES COMANDOS NA VPS (em ordem):

========================================
1️⃣ CONECTAR NA VPS
========================================
ssh seu-usuario@seu-ip-vps


========================================
2️⃣ IR PARA PASTA DO PROJETO
========================================
cd /caminho/do/agendapro-fullstack
# OU se não tem o projeto ainda:
# cd ~
# git clone https://github.com/Antonio1986-2025/agendapro-fullstack.git
# cd agendapro-fullstack


========================================
3️⃣ ATUALIZAR CÓDIGO (PULL)
========================================
git pull origin main


========================================
4️⃣ VER O QUE MUDOU
========================================
git log --oneline -5


========================================
5️⃣ VERIFICAR/ADICIONAR OPENAI_API_KEY NO .ENV
========================================
# Ver se já tem
grep OPENAI_API_KEY .env

# Se NÃO aparecer nada, adicione (cole SUA chave):
echo 'OPENAI_API_KEY=sk-proj-SUA_CHAVE_AQUI' >> .env

# Para editar o .env completo:
nano .env
# Adicione esta linha (com SUA chave OpenAI):
# OPENAI_API_KEY=sk-proj-...
# Salvar: Ctrl+O + Enter
# Sair: Ctrl+X


========================================
6️⃣ INSTALAR DEPENDÊNCIAS (SE NECESSÁRIO)
========================================
npm install --production


========================================
7️⃣ REINICIAR SERVIDOR
========================================

# SE USAR PM2 (mais comum):
pm2 restart all
pm2 logs --lines 50

# SE USAR DOCKER:
docker-compose restart app
docker-compose logs -f app

# SE USAR SYSTEMD:
sudo systemctl restart agendapro
sudo systemctl status agendapro

# SE ESTIVER RODANDO MANUALMENTE:
# Mate o processo: Ctrl+C
# Inicie de novo: npm start


========================================
8️⃣ TESTAR AGENTE IA
========================================
node test-ai.js


========================================
9️⃣ VER LOGS EM TEMPO REAL
========================================
# PM2
pm2 logs

# Docker
docker-compose logs -f app


========================================
🔟 TESTAR API
========================================
curl http://localhost:3000/api/health


========================================
✅ VERIFICAR SE ESTÁ RODANDO
========================================
# PM2
pm2 status

# Docker
docker-compose ps

# Manual
ps aux | grep node


========================================
📱 ATIVAR NO PAINEL WEB
========================================
1. Abra navegador: http://seu-dominio:3000
2. Login
3. Configurações → WhatsApp
4. Conectar WhatsApp (QR Code)
5. Ativar "Agente IA"
6. Salvar

========================================
🐛 SE DER PROBLEMA
========================================

# Ver erro completo nos logs
pm2 logs --lines 100 --err

# Reiniciar forçado
pm2 delete all
pm2 start npm --name "agendapro" -- start
pm2 save

# Ver uso de memória/CPU
pm2 monit

# Verificar porta
sudo lsof -i :3000


========================================
🔄 ROLLBACK (SE ALGO DEU MUITO ERRADO)
========================================
git log --oneline -5
git reset --hard <commit-hash-anterior>
pm2 restart all


========================================
📊 MONITORAMENTO CONTÍNUO
========================================
# Manter logs abertos
pm2 logs --raw

# Status a cada 2 segundos
watch -n 2 'pm2 status'
